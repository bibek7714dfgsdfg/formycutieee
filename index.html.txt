    <script>
        // Update to formatted clean calendar text
        const dtOptions = { year: 'numeric', month: 'long', day: 'numeric' };
        document.getElementById('live-date').innerText = new Date().toLocaleDateString('en-US', dtOptions);

        // --- ANIMATED ENVIRONMENT (SNOW + BURNING EMBERS) ---
        const snowCanvas = document.getElementById('snow-canvas');
        const sCtx = snowCanvas.getContext('2d');
        const emberCanvas = document.getElementById('ember-canvas');
        const eCtx = emberCanvas.getContext('2d');

        let width = snowCanvas.width = emberCanvas.width = window.innerWidth;
        let height = snowCanvas.height = emberCanvas.height = window.innerHeight;

        window.addEventListener('resize', () => {
            width = snowCanvas.width = emberCanvas.width = window.innerWidth;
            height = snowCanvas.height = emberCanvas.height = window.innerHeight;
        });

        // Snow setup
        const totalFlakes = 60;
        const flakesArray = [];
        for (let i = 0; i < totalFlakes; i++) {
            flakesArray.push({
                x: Math.random() * width, y: Math.random() * height,
                r: Math.random() * 3.5 + 1, d: Math.random() * totalFlakes,
                speed: Math.random() * 0.7 + 0.4
            });
        }

        // Fire Ember setup
        const totalEmbers = 35;
        const embersArray = [];
        for (let i = 0; i < totalEmbers; i++) {
            embersArray.push({
                x: Math.random() * width, y: height + Math.random() * 80,
                r: Math.random() * 2 + 0.5, speedY: Math.random() * 1.3 + 0.7,
                speedX: (Math.random() - 0.5) * 0.8, alpha: Math.random() * 0.8 + 0.2
            });
        }

        function runAnimationLoops() {
            // Render Falling Snow
            sCtx.clearRect(0, 0, width, height);
            sCtx.fillStyle = 'rgba(255, 255, 255, 0.8)';
            sCtx.beginPath();
            for (let i = 0; i < totalFlakes; i++) {
                let f = flakesArray[i];
                sCtx.moveTo(f.x, f.y);
                sCtx.arc(f.x, f.y, f.r, 0, Math.PI * 2, true);
                f.y += f.speed;
                f.x += Math.sin(f.d) * 0.35;
                if (f.y > height) {
                    flakesArray[i] = { x: Math.random() * width, y: 0, r: f.r, d: f.d, speed: f.speed };
                }
                f.d += 0.01;
            }
            sCtx.fill();

            // Render Rising Hot Embers
            eCtx.clearRect(0, 0, width, height);
            for (let i = 0; i < totalEmbers; i++) {
                let e = embersArray[i];
                eCtx.fillStyle = `rgba(255, ${Math.floor(Math.random() * 70 + 110)}, 10, ${e.alpha})`;
                eCtx.beginPath();
                eCtx.arc(e.x, e.y, e.r, 0, Math.PI * 2);
                eCtx.fill();

                e.y -= e.speedY;
                e.x += e.speedX;
                e.alpha -= 0.002;

                if (e.y < -10 || e.alpha <= 0) {
                    embersArray[i] = {
                        x: Math.random() * width, y: height + 10,
                        r: Math.random() * 2 + 0.5, speedY: Math.random() * 1.3 + 0.7,
                        speedX: (Math.random() - 0.5) * 0.8, alpha: Math.random() * 0.8 + 0.2
                    };
                }
            }

            requestAnimationFrame(runAnimationLoops);
        }
        runAnimationLoops();

        // --- INTERACTIVE FLOWER PLANTING ---
        const gardenContainer = document.getElementById('gardenContainer');
        let currentFlowers = 0;

        function addGardenFlower(emoji) {
            if (currentFlowers >= 12) {
                gardenContainer.querySelector('.sprouted-flower').remove();
                currentFlowers--;
            }

            const fl = document.createElement('div');
            fl.className = 'sprouted-flower';
            fl.innerText = emoji;

            const positionX = Math.floor(Math.random() * (gardenContainer.clientWidth - 40));
            const leanAngle = Math.floor(Math.random() * 24) - 12;

            fl.style.left = `${positionX}px`;
            fl.style.setProperty('--rot', `${leanAngle}deg`);

            fl.onclick = () => {
                fl.style.transform = `scale(1.35) rotate(${leanAngle}deg)`;
                setTimeout(() => {
                    fl.style.transform = `scale(1) rotate(${leanAngle}deg)`;
                }, 200);
            };

            gardenContainer.appendChild(fl);
            currentFlowers++;
        }

        // Add default romantic accents on startup
        window.addEventListener('DOMContentLoaded', () => {
            setTimeout(() => addGardenFlower('🌹'), 300);
            setTimeout(() => addGardenFlower('🔥'), 700);
            setTimeout(() => addGardenFlower('🌹'), 1100);
        });
